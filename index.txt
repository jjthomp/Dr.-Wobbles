    <script>
        function updateGuide() {
            // Get current selected values
            const sourcingOption = document.querySelector('input[name="sourcing"]:checked').value;
            const versionOption = document.querySelector('input[name="version"]:checked').value;

            // Update Sourcing Section
            if (sourcingOption === 'kit') {
                document.getElementById('sourcing-kit').classList.remove('hidden');
                document.getElementById('sourcing-diy').classList.add('hidden');
            } else {
                document.getElementById('sourcing-kit').classList.add('hidden');
                document.getElementById('sourcing-diy').classList.remove('hidden');
            }

            // Update Printing Section
            if (versionOption === 'standard') {
                document.getElementById('print-standard').classList.remove('hidden');
                document.getElementById('print-rechargeable').classList.add('hidden');
            } else {
                document.getElementById('print-standard').classList.add('hidden');
                document.getElementById('print-rechargeable').classList.remove('hidden');
            }
        }

        // Run once on load to ensure correct state
        updateGuide();
    </script>
